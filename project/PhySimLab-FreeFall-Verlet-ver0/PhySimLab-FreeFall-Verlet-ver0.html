Taylor series

\begin{align} f(x) & = f(a) + f^{\prime}(a)(x-a) + \frac{1}{2}f^{\prime\prime}(a)(x-a)^2 + \frac{1}{6}f^{\prime\prime\prime}(a)(x-a)^3 + \cdot\cdot\cdot \\ f(x) & = \sum_{n=0}^\infty \frac{f^{(n)}(a)}{n!} (x-a)^n \end{align}

Taylor series for x(t) and v(t)

\begin{align} x(t+\Delta t) & = x(t) + x^{\prime}(t)(t+\Delta t - t) + \frac{1}{2}x^{\prime\prime}(t)(t+\Delta t - t)^2 + \frac{1}{6}x^{\prime\prime\prime}(t)(t+\Delta t - t)^3 \\ x(t+\Delta t) & = x(t) + x^{\prime}(t)\Delta t + \frac{1}{2}x^{\prime\prime}(t)(\Delta t)^2 + \frac{1}{6}x^{\prime\prime\prime}(t)(\Delta t)^3 + \cdot\cdot\cdot \end{align}

\begin{align} v(t+\Delta t) & = v(t) + v^{\prime}(t)(t+\Delta t - t) + \frac{1}{2}v^{\prime\prime}(t)(t+\Delta t - t)^2 + \frac{1}{6}v^{\prime\prime\prime}(t)(t+\Delta t - t)^3 \\ v(t+\Delta t) & = v(t) + v^{\prime}(t)\Delta t + \frac{1}{2}v^{\prime\prime}(t)(\Delta t)^2 + \frac{1}{6}v^{\prime\prime\prime}(t)(\Delta t)^3 + \cdot\cdot\cdot \end{align}

\begin{align} \because v(t) & =\frac{{\rm d}x(t)}{{\rm d}t} \\ a(t) & =\frac{{\rm d}v(t)}{{\rm d}t} \end{align}

\begin{align} \therefore x(t+\Delta t) & = x(t) + v(t)\Delta t + \frac{1}{2}a(t)(\Delta t)^2 + \frac{1}{6}x^{\prime\prime\prime}(t)(\Delta t)^3 + \cdot\cdot\cdot \\ v(t+\Delta t) & = v(t) + a(t)\Delta t + \frac{1}{2}v^{\prime\prime}(t)(\Delta t)^2 + \frac{1}{6}v^{\prime\prime\prime}(t)(\Delta t)^3 + \cdot\cdot\cdot \end{align}

Verlet integration

\begin{align} x(t+\Delta t) & = x(t) + v(t)\Delta t + \frac{1}{2}a(t)(\Delta t)^2 + \frac{1}{6}x^{\prime\prime\prime}(t)(\Delta t)^3 + O((\Delta t)^4) \\ x(t+\Delta t) & = x(t) - v(t)\Delta t + \frac{1}{2}a(t)(\Delta t)^2 - \frac{1}{6}x^{\prime\prime\prime}(t)(\Delta t)^3 + O((\Delta t)^4) \end{align}

\begin{align} x(t+\Delta t) + x(t- \Delta t) & = 2x(t) + a(t)(\Delta t)^2 + O((\Delta t)^4) \\ x(t+\Delta t) & = 2x(t) - x(t- \Delta t) + a(t)(\Delta t)^2 + O((\Delta t)^4) \end{align}

Finite difference method

\[x_{n+1} = 2x_n - x_{n-1} + a_n(\Delta t)^2 + O((\Delta t)^4) \]

\[v_{n+1} = v_{n-1} + 2a_n\Delta t + O((\Delta t)^3) \]

Initial status

\[x_{-1} = x_0 - v_0\Delta t + \frac{1}{2}a_0(\Delta t)^2 + O((\Delta t)^3) \]

\[v_{-1} = v_0 - a_0\Delta t + O((\Delta t)^2) \]
